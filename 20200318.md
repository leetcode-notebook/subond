#### 位1的个数



```go
// date 2020/03/18
// 算法：移位比较
func hammingWeight(num uint32) int {
    var c int
    for num > 0 {
        c += int(num & 0x1)
        num >>= 1
    }
    return c
}
```



#### 汉明距离

```go
// date 2020/03/18
func hammingDistance(x int, y int) int {
    return hammingWeight(x^y)
}

func hammingWeight(num int) int {
    var c int
    for num > 0 {
        c += num & 0x1
        num >>= 1
    }
    return c
}
```
